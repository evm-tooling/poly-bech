================================================================================
Simple after block with speedup chart
================================================================================

use std::charting

suite test {
    bench foo {
        go: run()
    }

    after {
        charting.drawSpeedupChart(
            title: "Performance"
        )
    }
}

================================================================================
Multiple chart directives
================================================================================

use std::charting

suite test {
    bench foo {
        go: run()
    }

    after {
        charting.drawSpeedupChart(
            title: "Speedup Chart",
            output: "speedup.svg"
        )

        charting.drawTable(
            title: "Table",
            sortBy: "name"
        )
    }
}

================================================================================
Chart with all parameter types
================================================================================

use std::charting

suite test {
    bench foo {
        go: run()
    }

    after {
        charting.drawSpeedupChart(
            title: "Full Example",
            width: 800,
            height: 600,
            minSpeedup: 1.5,
            includeBenchmarks: ["foo", "bar"],
            sortBy: "speedup"
        )
    }
}
